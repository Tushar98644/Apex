cmake_minimum_required(VERSION 3.12)
project(trading_engine)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# Find pybind11
find_package(pybind11 REQUIRED)

# Include existing directories
include_directories(../include)
include_directories(../libraries)

# Find CURL
find_package(PkgConfig REQUIRED)
pkg_check_modules(CURL REQUIRED libcurl)

# Find Boost
find_package(Boost REQUIRED COMPONENTS system)

# source files
set(COMMON_SOURCES
    ../src/api_client.cpp
    ../src/config.cpp
    ../src/order_manager.cpp
    ../src/logger.cpp
    ../src/websocket_server.cpp
    ../src/latency_benchmark.cpp
)

# Create the Python module
pybind11_add_module(trading_engine 
    ../src/python_bindings.cpp
    ${COMMON_SOURCES}
)

# Link all existing libraries
target_link_libraries(trading_engine PRIVATE 
    ${CURL_LIBRARIES}
    ${Boost_LIBRARIES}
)

# Include directories for dependencies
target_include_directories(trading_engine PRIVATE 
    ${CURL_INCLUDE_DIRS}
    ${Boost_INCLUDE_DIRS}
)

# Compiler flags
target_compile_options(trading_engine PRIVATE 
    -O3 
    -march=native
    ${CURL_CFLAGS_OTHER}
)
